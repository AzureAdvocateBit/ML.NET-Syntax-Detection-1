def send_sms(url, values):
    postdata = urllib.urlencode(values)
    req = urllib2.Request(url, postdata)
    try:
        response = urllib2.urlopen(req)
        response_url = response.geturl()
        if response_url == url:
            print(
                'SMS SENT \nTo: {}\nFrom: {}\nMessage: {}\nUsername: {}'.format(
                    values['selectednums'],
                    values['from'],
                    values['message'],
                    values['uname'],
                )
            )
    except urllib2.URLError, e:
        print(
            'SMS FAILED \nTo: {}\nFrom: {}\nMessage: {}\nUsername: {}'.format(
                values['selectednums'],
                values['from'],
                values['message'],
                values['uname'],
            )
        )
